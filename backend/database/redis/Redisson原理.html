                            那么拿到这个锁的剩余有效时间有什么用呢？我们回到之前的 tryLock() 方法

                            	@Override
    public boolean tryLock(long waitTime, long leaseTime, TimeUnit unit) throws InterruptedException {
        // 将等待时间转化为毫秒
        long time = unit.toMillis(waitTime);
        // 当前时间
        long current = System.currentTimeMillis();
        // 当前线程id
        long threadId = Thread.currentThread().getId();
        // 锁的剩余时间，tryAcquire() 获取锁失败时返回的是锁的过期时间
        Long ttl = tryAcquire(waitTime, leaseTime, unit, threadId);
        // lock acquired
        if (ttl == null) {
            // 获取锁成功
            return true;
        }
        // 剩下的等待时间
        time -= System.currentTimeMillis() - current;
        if (time <= 0) {
            // 获取锁失败
            acquireFailed(waitTime, unit, threadId);
            return false;
        }
        
        current = System.currentTimeMillis();
        // 订阅消息
        RFuture<RedissonLockEntry> subscribeFuture = subscribe(threadId);
        // 设置最大的等待时间为 time
        if (!subscribeFuture.await(time, TimeUnit.MILLISECONDS)) {
            if (!subscribeFuture.cancel(false)) {
                subscribeFuture.onComplete((res, e) -> {
                    if (e == null) {
                        // 在 time 时间内没有收到消息，取消订阅
                        unsubscribe(subscribeFuture, threadId);
                    }
                });
            }
            // 获取锁失败
            acquireFailed(waitTime, unit, threadId);
            return false;
        }

        try {
            // 剩下的等待时间
            time -= System.currentTimeMillis() - current;
            if (time <= 0) {
                acquireFailed(waitTime, unit, threadId);
                return false;
            }
            while (true) {
                long currentTime = System.currentTimeMillis();
                ttl = tryAcquire(waitTime, leaseTime, unit, threadId);
                // lock acquired
                if (ttl == null) {
                    return true;
                }

                time -= System.currentTimeMillis() - currentTime;
                if (time <= 0) {
                    // 获取锁失败
                    acquireFailed(waitTime, unit, threadId);
                    return false;
                }

                // waiting for message
                currentTime = System.currentTimeMillis();
                if (ttl >= 0 && ttl < time) {
                    // 使用信号量方式
                    // 锁还没有过期并且过期时间小于等待时间，我们就设置获取锁的等待时间为 ttl
                    // 因为如果锁释放了就没必要再等 time 时间了
                    subscribeFuture.getNow().getLatch().tryAcquire(ttl, TimeUnit.MILLISECONDS);
                } else {
                    // 否则设置获取锁的等待时间为 time
                    subscribeFuture.getNow().getLatch().tryAcquire(time, TimeUnit.MILLISECONDS);
                }
                time -= System.currentTimeMillis() - currentTime;
                if (time <= 0) {
                    // 获取锁失败
                    acquireFailed(waitTime, unit, threadId);
                    return false;
                }
                // 若等待时间还有剩余，会再次进入到 while 循坏体内
            }
        } finally {
            // 取消订阅
            unsubscribe(subscribeFuture, threadId);
        }
//        return get(tryLockAsync(waitTime, leaseTime, unit));
    }

                            • Redisson锁重试在第一次获取锁失败后，会通过发布订阅方式，等待锁释放后再去获取锁
                            • Redisson锁重试在第二次获取锁失败后，会通过信号量方式，等待锁释放后再去获取锁

                            private <T> RFuture<Long> tryAcquireAsync(long waitTime, long leaseTime, TimeUnit unit, long threadId) {
    if (leaseTime != -1) {
        return tryLockInnerAsync(waitTime, leaseTime, unit, threadId, RedisCommands.EVAL_LONG);
    }
    RFuture<Long> ttlRemainingFuture = tryLockInnerAsync(waitTime,
                                                         commandExecutor.getConnectionManager().getCfg().getLockWatchdogTimeout(), // 设置看门狗时间
                                                         TimeUnit.MILLISECONDS, threadId, RedisCommands.EVAL_LONG);
    // 回调函数
    // ttlRemaining：锁的剩余有效期
    // e：异常
    ttlRemainingFuture.onComplete((ttlRemaining, e) -> {
        if (e != null) {
            // 有异常，直接返回
            return;
        }

        // lock acquired
        if (ttlRemaining == null) {
            // 获取锁成功，进行续约逻辑
            scheduleExpirationRenewal(threadId);
        }
    });
    return ttlRemainingFuture;
}

                            scheduleExpirationRenewal()

                            静态的 ConcurrentMap，key 是 String 类型，value 是 ExpirationEntry 类型

                            getEntryName()

                            • id 是当前连接 id
                            • name 是锁的名称

                            getEntryName() 就是来获取锁的名称

                            private void scheduleExpirationRenewal(long threadId) {
        ExpirationEntry entry = new ExpirationEntry();
    	// 把 entry 放到 map 中，一个锁对应一个 entry
    	// 保证不管重入多少次，一个锁总是拿到相同的 entry
        ExpirationEntry oldEntry = EXPIRATION_RENEWAL_MAP.putIfAbsent(getEntryName(), entry);
    	// 不是第一次执行
        if (oldEntry != null) {
            // 第二次执行的时候，entry 中已经有定时任务去重置锁的过期时间
            // 把 threadId 加入到 entry 中，体现可重入的思想
            oldEntry.addThreadId(threadId);
        } else {
            // 第一次执行
            // 把 threadId 加入到 entry 中
            entry.addThreadId(threadId);
            // 续约，更新有效期
            renewExpiration();
        }
    }

                            renewExpiration()

                            // 把定时任务加入到 entry 中
private void renewExpiration() {
    	// 先获取 entry
        ExpirationEntry ee = EXPIRATION_RENEWAL_MAP.get(getEntryName());
        if (ee == null) {
            return;
        }
        // 开启一个定时任务
        Timeout task = commandExecutor.getConnectionManager().newTimeout(new TimerTask() {
            @Override
            public void run(Timeout timeout) throws Exception {
                // 保证同一把锁获取到的是同一个 entry
                ExpirationEntry ent = EXPIRATION_RENEWAL_MAP.get(getEntryName());
                if (ent == null) {
                    return;
                }
                // 获取线程iD
                Long threadId = ent.getFirstThreadId();
                if (threadId == null) {
                    return;
                }
                // 设置锁的过期时间
                RFuture<Boolean> future = renewExpirationAsync(threadId);
                // 回调函数
                future.onComplete((res, e) -> {
                    if (e != null) {
                        // 有异常
                        log.error("Can't update lock " + getName() + " expiration", e);
                        return;
                    }
                    
                    if (res) {
                        // reschedule itself
                        // 递归调用当前方法
                        renewExpiration();
                    }
                });
            }
        // delay: 默认是 internalLockLeaseTime / 3 之后执行
        }, internalLockLeaseTime / 3, TimeUnit.MILLISECONDS);
        // 把定时任务加入到 entry 中，也就是 entry 现在有 threadId，定时任务 task
        ee.setTimeout(task);
    }

                            protected RFuture<Boolean> renewExpirationAsync(long threadId) {
        return evalWriteAsync(getName(), LongCodec.INSTANCE, RedisCommands.EVAL_BOOLEAN,
                // 判断锁是否是自己的

                        // 设置过期时间
                        "redis.call('pexpire', KEYS[1], ARGV[1]); " +
                        "return 1; " +
                        "end; " +

                            总结：

                            • 获取锁成功之后会执行一个回调函数，进行续约逻辑

                            • renewExpiration()方法开启了一个延时执行的定时任务，定时任务就是去重置锁的有效期，默认是每隔 10s(看门狗时间 / 3) 执行一次，执行完后又递归调用自己，不断续约

                            • ExpirationEntry 存放 线程ID 和 延时执行的定时任务

    	// 根据锁名称得到 entry
        ExpirationEntry task = EXPIRATION_RENEWAL_MAP.get(getEntryName());
        if (task == null) {
            return;
        }
  
        if (threadId != null) {
            // 移除线程id
            task.removeThreadId(threadId);
        }

        if (threadId == null || task.hasNoThreads()) {
            Timeout timeout = task.getTimeout();
            if (timeout != null) {
                // 取消定时任务
                timeout.cancel();
            }
            // 移除 entry
            EXPIRATION_RENEWAL_MAP.remove(getEntryName());
        }
    }

                            # 总结

                            Redisson分布式锁原理：

                            • 可重入：利用hash结构记录线程id和重入次数

                              每次获取锁的时候，如果获取失败，判断下锁的标识，如果是自己的话，重入次数加一，释放锁的时候，重入次数减一，直到减为 0

                            • 可重试：利用信号量和PubSub功能实现等待、唤醒，获取锁失败的重试机制

                              获取锁失败的时候，不会马上又去获取锁，而是去等待一个释放锁的消息，当我们获取锁成功的线程释放锁的时候，会发布一个释放锁的消息，当我们得到消息以后，就会重新获取锁，获取锁失败，会继续等待锁释放的信号，继续重试，但这个重试不是无限的重试，会有一个等待时间，当超过这个等待时间，就不再等待。因为采用这种等待唤醒的机制，所以不会占用很多的 CPU 资源

                            • 超时续约：利用watchDog，每隔一段时间（releaseTime / 3），重置超时时间

                              获取锁成功之后，会开启一个定时任务，每隔一段时间就去重置锁的有效时间

                            # 3.multiLock联锁

                            redis只有一个节点的情况下，一旦宕机，然后锁就失效了，因此我们需要 联锁 来保证分布式锁的一致性。

                            假设我们有3个redis节点，没有从节点。

                            tryLock()

                            @Override
    public boolean tryLock(long waitTime, long leaseTime, TimeUnit unit) throws InterruptedException {
//        try {
//            return tryLockAsync(waitTime, leaseTime, unit).get();
//        } catch (ExecutionException e) {
//            throw new IllegalStateException(e);
//        }
        long newLeaseTime = -1;
        if (leaseTime != -1) {
            if (waitTime == -1) {
               	// 不可重试，那么 newLeaseTime 就是原来的 leaseTime
                newLeaseTime = unit.toMillis(leaseTime);
            } else {
                // 可重试，newLeaseTime 为 waitTime 的两倍，避免在重试的时候锁过期了
                newLeaseTime = unit.toMillis(waitTime)*2;
            }
        }
        
        long time = System.currentTimeMillis();
        long remainTime = -1;
        if (waitTime != -1) {
           	// 可重试，remainTime 就是剩余的等待时间
            remainTime = unit.toMillis(waitTime);
        }
        // 等于 remainTime
        long lockWaitTime = calcLockWaitTime(remainTime);
        // 返回 0
        int failedLocksLimit = failedLocksLimit();
        // 存放已经获取到的锁的集合
        List<RLock> acquiredLocks = new ArrayList<>(locks.size());
        // 遍历锁
        for (ListIterator<RLock> iterator = locks.listIterator(); iterator.hasNext();) {
            RLock lock = iterator.next();
            boolean lockAcquired;
            try {
                if (waitTime == -1 && leaseTime == -1) {
                    // 尝试获取锁，只获取一次
                    lockAcquired = lock.tryLock();
                } else {
                    long awaitTime = Math.min(lockWaitTime, remainTime);
                    // 尝试获取锁，可重试
                    lockAcquired = lock.tryLock(awaitTime, newLeaseTime, TimeUnit.MILLISECONDS);
                }
            } catch (RedisResponseTimeoutException e) {
                unlockInner(Arrays.asList(lock));
                lockAcquired = false;
            } catch (Exception e) {
                lockAcquired = false;
            }
            
            if (lockAcquired) {
                // 获取一个锁成功，添加到集合中
                acquiredLocks.add(lock);
            } else {
                // 获取锁失败
                // 假设第一个锁就获取失败，总共是3个锁， 3 - 0 != 0
                if (locks.size() - acquiredLocks.size() == failedLocksLimit()) {
                    // 当获取到所有的锁后直接跳出循坏
                    break;
                }

                if (failedLocksLimit == 0) {
                    // 释放掉已经获取到的锁
                    unlockInner(acquiredLocks);
                    if (waitTime == -1) {
                        // 不可重试，返回 false
                        return false;
                    }
                    // 0
                    failedLocksLimit = failedLocksLimit();
                    // 清空获取锁集合
                    acquiredLocks.clear();
                    // reset iterator
                    while (iterator.hasPrevious()) {
                        // 将迭代器指向第一个位置
                        iterator.previous();
                    }
                } else {
                    failedLocksLimit--;
                }
            }
            
            if (remainTime != -1) {
                remainTime -= System.currentTimeMillis() - time;
                time = System.currentTimeMillis();
                if (remainTime <= 0) {
                    // 已经超时，释放掉已经获取到的锁
                    unlockInner(acquiredLocks);
                    return false;
                }
            }
            // 不会超时，继续进入 for 循坏获取锁
        }

        // 获取所有的锁成功并且设置了过期时间
        // 没有设置过期时间的话走的是 watchDog 机制，建议不设置过期时间
        if (leaseTime != -1) {
          	// 设置了锁的过期时间
            List<RFuture<Boolean>> futures = new ArrayList<>(acquiredLocks.size());
            for (RLock rLock : acquiredLocks) {
                // 遍历所有的锁，重置过期时间，目的是让所有的锁的过期时间保持一致
                RFuture<Boolean> future = ((RedissonLock) rLock).expireAsync(unit.toMillis(leaseTime), TimeUnit.MILLISECONDS);
                futures.add(future);
            }
            
            for (RFuture<Boolean> rFuture : futures) {
                rFuture.syncUninterruptibly();
            }
        }
        
        return true;
    }

                            # 4.总结

                            1）不可重入Redis分布式锁：

                            原理：利用setnx的互斥性；利用ex避免死锁；释放锁时判断线程标示

                            缺陷：不可重入、无法重试、锁超时失效

                            2）可重入的Redis分布式锁：

                            原理：利用hash结构，记录线程标示和重入次数；利用watchDog延续锁时间；利用信号量控制锁重试等待

                            缺陷：redis宕机引起锁失效问题

                            3）Redisson的multiLock：

                            原理：多个独立的Redis节点，必须在所有节点都获取重入锁，才算获取锁成功

                            缺陷：运维成本高、实现复杂
