                                                                                                      # 2. 两数相加open in new window

                                                                                                      给你两个 非空 的链表，表示两个非负的整数。它们每位数字都是按照 逆序 的方式存储的，并且每个节点只能存储 一位 数字。

                                                                                                      请你将两个数相加，并以相同形式返回一个表示和的链表。

                                                                                                      你可以假设除了数字 0 之外，这两个数都不会以 0 开头。

                                                                                                      解法一（可不看）：

                                                                                                      class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode p1 = l1, p2 = l2;
        ListNode pre = null;
        // 1.同时遍历两条链表
        while (p1 != null && p2 != null) {
            // 2.相加
            int val = p1.val + p2.val;
            p1.val = val % 10;
            // 3.判断是否产生进位
            if (val >= 10) {
                ListNode p = p1.next;
                if (p == null) {
                    ListNode temp = new ListNode(1);
                    p1.next = temp;
                } else {
                    p.val++;
                    // 判断是否产生进位
                    while (p.val >= 10) {
                        p.val %= 10;
                        if (p.next != null) {
                            p.next.val++;
                        } else {
                            p.next = new ListNode(1);
                        }
                        p = p.next;
                    }
                }
            }
            pre = p1;
            p1 = p1.next;
            p2 = p2.next;
        }
        // l2链表比较长
        if (p1 == null) {
            pre.next = p2;
        }
        // 4.返回结果
        return l1;
    }

                                                                                                      https://leetcode.cn/problems/add-two-numbers/solution/hua-jie-suan-fa-2-liang-shu-xiang-jia-by-guanpengc/open in new window

                                                                                                      把两条链表看成相等长度，null 用 0 代替

                                                                                                      1.创建虚拟头节点

                                                                                                      2.遍历两条链表（只要其中有一个不为空）

                                                                                                      3.判断是否为空，用 0 代替

                                                                                                      4.计算，带上进位

                                                                                                      5.更新计算结果和进位，更新链表

                                                                                                      6.指针移动

                                                                                                      7.判断进位是否大于0，是的话再创建节点

                                                                                                      // 时间和空间 O(max(m, n)) 
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        // 1.虚拟头节点
        ListNode dummy = new ListNode(-1);
        ListNode p = dummy, p1 = l1, p2 = l2;
        // 进位
        int carry = 0;
        // 2.遍历两条链表，看作是相等长度的
        while (p1 != null || p2 != null) {
            // 3.null当做0
            int val1 = p1 == null ? 0 : p1.val;
            int val2 = p2 == null ? 0 : p2.val;
            // 4.相加考虑进位
            int sum = val1 + val2 + carry;
            carry = sum / 10;
            sum %= 10;
            p.next = new ListNode(sum);
            // 5.指针移动
            p = p.next;
            if (p1 != null) p1 = p1.next;
            if (p2 != null) p2 = p2.next;
        }
        // 6.判断进位是否大于0，创建新节点
        if (carry > 0) {
            p.next = new ListNode(1);
        }
        return dummy.next;
    }
}

                                                                                                      # 4. 寻找两个正序数组的中位数open in new window

                                                                                                      给定两个大小分别为 m 和 n 的正序（从小到大）数组 nums1 和 nums2。请你找出并返回这两个正序数组的 中位数 。

                                                                                                      算法的时间复杂度应该为 O(log (m+n)) 。

                                                                                                      1.合并两个数组

                                                                                                      2.判断奇偶数，返回中位数

                                                                                                      // 时间复杂度和空间复杂度都是 O(m + n)
// 遍历了两个数组
class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        // 1.合并两个数组
        int m = nums1.length, n = nums2.length;
        int len = m + n;
        int[] temp = new int[len];
        int i = 0, j = 0;
        for (int k = 0; k < temp.length; k++) {
            if (i == m) {
                temp[k] = nums2[j++];
            } else if (j == n) {
                temp[k] = nums1[i++];
            } else if (nums1[i] < nums2[j]) {
                temp[k] = nums1[i++];
            } else {
                temp[k] = nums2[j++];
            }
        }
        // 2.返回中位数
        if (len % 2 == 1) {
            return temp[len / 2];
        } 
        return (temp[len / 2 - 1] + temp[len / 2]) / 2.0;
    }
}

                                                                                                      # 17. 电话号码的字母组合open in new window

                                                                                                      给定一个仅包含数字 2-9 的字符串，返回所有它能表示的字母组合。答案可以按 任意顺序 返回。

                                                                                                      给出数字到字母的映射如下（与电话按键相同）。注意 1 不对应任何字母。

                                                                                                      回溯算法，计算组合

                                                                                                      1.当前可选择的列表 digits[start] 对应的字母

                                                                                                      2.结束条件

                                                                                                      3.路径

                                                                                                      // 时间 O(2^n) 
class Solution {

    // 记录路径
    StringBuilder track = new StringBuilder();
    // 记录结果
    LinkedList<String> res = new LinkedList<>();
    // 记录映射关系
    Map<Character, String> map;

    public List<String> letterCombinations(String digits) {
        // base case
        if (digits.length() == 0) {
            return res;
        }
        map = new HashMap<>();
        map.put('2', "abc");
        map.put('3', "def");
        map.put('4', "ghi");
        map.put('5', "jkl");
        map.put('6', "mno");
        map.put('7', "pqrs");
        map.put('8', "tuv");
        map.put('9', "wxyz");
        backtrack(digits, 0);
        return res;
    }

    public void backtrack(String digits, int start) {
        // 结束条件
        if (digits.length() == track.length()) {
            res.add(track.toString());
            return;
        }
        String val = map.get(digits.charAt(start));
        for (int i = 0; i < val.length(); i++) {
            char ch = val.charAt(i);
            // 选择
            track.append(ch);
            // 进入下一层决策树
            backtrack(digits, start + 1);
            // 撤销选择 
            track.deleteCharAt(track.length() - 1);
        }
    }
}

                                                                                                      # 31. 下一个排列open in new window

                                                                                                      整数数组的一个 排列 就是将其所有成员以序列或线性顺序排列。

                                                                                                      例如，arr = [1,2,3] ，以下这些都可以视作 arr 的排列：[1,2,3]、[1,3,2]、[3,1,2]、[2,3,1] 。 整数数组的 下一个排列 是指其整数的下一个字典序更大的排列。更正式地，如果数组的所有排列根据其字典顺序从小到大排列在一个容器中，那么数组的 下一个排列 就是在这个有序容器中排在它后面的那个排列。如果不存在下一个更大的排列，那么这个数组必须重排为字典序最小的排列（即，其元素按升序排列）。

                                                                                                      例如，arr = [1,2,3] 的下一个排列是 [1,3,2] 。 类似地，arr = [2,3,1] 的下一个排列是 [3,1,2] 。 而 arr = [3,2,1] 的下一个排列是 [1,2,3] ，因为 [3,2,1] 不存在一个字典序更大的排列。 给你一个整数数组 nums ，找出 nums 的下一个排列。

                                                                                                      必须 原地 修改，只允许使用额外常数空间。

                                                                                                      https://leetcode.cn/problems/next-permutation/solution/xia-yi-ge-pai-lie-suan-fa-xiang-jie-si-lu-tui-dao-/open in new window

                                                                                                      1.从后往前找出相邻的升序的两个元素 nums[i] nums[j], 此时 [j, end] 是降序的

                                                                                                      2.找不到，跳到第五步

                                                                                                      3.在 [j,end] 中，从后往前找出第一个大于 nums[i] 的数 nums[k]

                                                                                                      4.交换这两个数

                                                                                                      5.升序 [j, end]

                                                                                                      解法一：

                                                                                                      class Solution {
    public void nextPermutation(int[] nums) {
        int n  = nums.length;
        // 1.从后往前找出相邻的升序的两个元素 nums[i] nums[j], 此时 [j, end] 是降序的
        int j = -1;
        for (int i = n - 1; i >= 1; i--) {
            if (nums[i - 1] < nums[i]) {
                j = i;
                break;
            }
        }
        // 2.找不到，直接升序
        if (j == -1) {
            Arrays.sort(nums);
            return;
        }
        // 3.在 [j,end] 中，从后往前找出第一个大于 nums[i] 的数 nums[k]
        int k = -1;
        for (int i = n - 1; i >= j; i--) {
            if (nums[i] > nums[j - 1]) {
                k = i;
                break;
            }
        }
        // 4.交换这两个数
        swap(nums, j - 1, k);
        // 5.升序 [j, end]   
        Arrays.sort(nums, j, n); // 左闭右开
    }

    public void swap(int[] nums, int i, int j) {
        int t = nums[i];
        nums[i] = nums[j];
        nums[j] = t;
    }
}

    // 从后往前找到第一个升序的 nums[i] < nums[j]，对[j...n]升序，在[j...n] 中找到第一个大于 nums[i] 的数，交换

    public void nextPermutation(int[] nums) {
        int n = nums.length;
        for (int i = n - 2; i >= 0; i--) {
            if (nums[i] < nums[i + 1]) {
                Arrays.sort(nums, i + 1, n);
                for (int j = i + 1; j < n; j++) {
                    if (nums[i] < nums[j]) {
                        int t = nums[i];
                        nums[i] = nums[j];
                        nums[j] = t; 
                        return;
                    }
                }
            }
        }
        Arrays.sort(nums);
    }
}

                                                                                                      # 32. 最长有效括号open in new window

                                                                                                      给你一个只包含 '(' 和 ')' 的字符串，找出最长有效（格式正确且连续）括号子串的长度。

                                                                                                      class Solution {
    public int longestValidParentheses(String s) {
        // 记录左括号的索引
        Stack<Integer> sta = new Stack<>();
        int n  = s.length();
        // dp[i] 表示以 s[i - 1] 结尾的有效括号的长度
        int[] dp = new int[n + 1];
        for (int i = 0; i < n; i++) {
            char ch = s.charAt(i);
            if (ch == '(') {
                // 左括号索引入栈
                sta.push(i);
                // 左括号结尾肯定不是有效括号
                dp[i + 1] = 0;
            } else {
                // 右括号
                if (!sta.isEmpty()) {
                    // 得到左括号的索引
                    int leftIndex = sta.pop();
                    // 计算有效括号长度
                    int len = i - leftIndex + 1 + dp[leftIndex];
                    // 更新dp数组
                    dp[i + 1] = len;
                } else {
                    // 无可用的左括号
                    dp[i + 1] = 0;
                }
            }
        }
        // 找到最长的有效括号
        int res = Integer.MIN_VALUE;
        for (int i = 0; i <= n; i++) {
            res = Math.max(res, dp[i]);
        }
        return res;
    }
}

                                                                                                      # 33. 搜索旋转排序数组open in new window

                                                                                                      整数数组 nums 按升序排列，数组中的值 互不相同 。

                                                                                                      在传递给函数之前，nums 在预先未知的某个下标 k（0 <= k < nums.length）上进行了 旋转，使数组变为 [nums[k], nums[k+1], ..., nums[n-1], nums[0], nums[1], ..., nums[k-1]]（下标 从 0 开始 计数）。例如， [0,1,2,4,5,6,7] 在下标 3 处经旋转后可能变为 [4,5,6,7,0,1,2] 。

                                                                                                      给你 旋转后 的数组 nums 和一个整数 target ，如果 nums 中存在这个目标值 target ，则返回它的下标，否则返回 -1 。

                                                                                                      你必须设计一个时间复杂度为 O(log n) 的算法解决此问题。

                                                                                                      将数组一分为二，其中一定有一个是有序的，另一个可能是有序，也能是部分有序。

                                                                                                      此时有序部分用二分法查找。无序部分再一分为二，其中一个一定有序，另一个可能有序，可能无序。就这样循环.

                                                                                                      关键是要分清楚有序数组和旋转数组的边界

                                                                                                      复杂度分析

                                                                                                      • 时间复杂度 O(logn)
                                                                                                      • 空间复杂度 O(1)
                                                                                                      class Solution {
    public int search(int[] nums, int target) {
        int lo = 0, hi = nums.length - 1;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            // 找到直接返回
            if (nums[mid] == target) return mid;
            // 注意：这里是 <= 
            // 例子：[3, 1], lo = 0, mid = 0, hi = 1,如果是 < ，会认为 [mid, hi] 是有序数组，不符合
            // [lo, mid] 是有序数组，[mid + 1, hi] 是旋转数组
            if (nums[lo] <= nums[mid]) {
                // 前半部分有序
                if (nums[lo] <= target && target < nums[mid]) {
                    // target 在前半部分
                    hi = mid - 1;
                } else {
                    // 去后半部分找
                    lo = mid + 1;
                }
            // [lo, mid - 1] 是旋转数组，[mid, hi] 是有序数组
            } else {
                // 后半部分有序
                if (nums[mid] < target && target <= nums[hi]) {
                    // target 在后半部分
                    lo = mid + 1;
                } else {
                    // 去前半部分找
                    hi = mid - 1;
                }
            }
        }
        return -1;
    }
}

                                                                                                      # 49. 字母异位词分组open in new window

                                                                                                      给你一个字符串数组，请你将 字母异位词 组合在一起。可以按任意顺序返回结果列表。

                                                                                                      字母异位词 是由重新排列源单词的字母得到的一个新单词，所有源单词中的字母通常恰好只用一次。

                                                                                                      如何判断两个字符串是字母异位词？

                                                                                                      对字符串进行编码，让字母异位词的字符串返回相同的编码。具体我们可以记录每个字符出现的次数，返回编码字符串

                                                                                                      class Solution {
    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String, List<String>> map = new HashMap<>();
        // 1.遍历字符串，将相同编码值的字符串放在一起
        for (String str : strs) {
            String code = encode(str);
            map.putIfAbsent(code, new ArrayList<>());
            map.get(code).add(str);
        }
        // 2.遍历map的值，加入res，返回结果
        List<List<String>> res = new ArrayList<>();
        for (List<String> list : map.values()) {
            res.add(list);
        }
        return res;
    }

    // 对字符串进行编码，记录每个字符出现的个数，返回字符串
    public String encode(String s) {
        // 注意：这里是 char[]，不使用 int[] 的原因是 int[] 没有重写 toString() 方法
        char[] count = new char[26];
        for(char c : s.toCharArray()) {
            count[c - 'a']++;
        }
        return new String(count);
    }
}

                                                                                                      # 62. 不同路径open in new window

                                                                                                      一个机器人位于一个 m x n 网格的左上角 （起始点在下图中标记为 “Start” ）。

                                                                                                      机器人每次只能向下或者向右移动一步。机器人试图达到网格的右下角（在下图中标记为 “Finish” ）。

                                                                                                      问总共有多少条不同的路径？

    public int uniquePaths(int m, int n) {
        // dp[i][j] 表示 [0,0] 移动到 [i,j] 有多少条路径
        int[][] dp = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                // base case: i == 0 || j == 0, dp[0][j] = 1
                if (i == 0 || j == 0) {
                    dp[i][j] = 1;
                    continue;
                }
                // 做选择
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
        return dp[m - 1][n - 1];
    }
}

        int p0 = 0, p1 = 0;

                p1++;
            }
        }

    boolean[][] visited;

    // dfs+回溯
    public boolean exist(char[][] board, String word) {
        int m = board.length, n = board[0].length;
        visited = new boolean[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                // 从起点开始dfs
                if (dfs(board, i, j, word, 0)) {
                    return true;
                }
            }
        }
        return false;
    }

    public boolean dfs(char[][] board, int i, int j, String word, int k) {
        if (k == word.length()) {
            // word存在，直接返回
            return true;
        }
        int m = board.length, n = board[0].length;
        // 检查边界
        if (i < 0 || j < 0 || i >= m || j >= n) return false;
        // 已经遍历过 或者 元素不相等
        if (visited[i][j] || board[i][j] != word.charAt(k)) return false;
        // 做选择
        visited[i][j] = true;
        // dfs 上下左右
        if (dfs(board, i + 1, j, word, k + 1)) return true;
        if (dfs(board, i - 1, j, word, k + 1)) return true;
        if (dfs(board, i, j + 1,  word, k + 1)) return true;
        if (dfs(board, i , j - 1, word, k + 1)) return true;
        // 撤销选择
        visited[i][j] = false;
        // 走到这里，说明不匹配，返回false
        return false;
    }
}

    // 第 i 个位置的最大矩形的面积是什么呢？
    // 往左边找到第一个小于 heights[i] 的索引 left，往右边找到第一个小于 height[i] 的索引 right
    // 最大面积 = height[i] * (right - left - 1)
    public int largestRectangleArea(int[] heights) {
       int n = heights.length;
       // 向数组首尾各添加一个0，保证可以找到 left，right
       int[] new_heights = new int[n + 2];
       for (int i = 1; i < n + 1; i++) {
           new_heights[i] = heights[i - 1];
       } 
       Stack<Integer> sta = new Stack<>();
       int res = 0;
       for (int i = 0; i < new_heights.length; i++) {
           // 单调递增栈
           // 注意这里是 < 
           while (!sta.isEmpty() && new_heights[i] < new_heights[sta.peek()]) {
                // 当前元素的索引
                int cur = sta.pop();
                // 左边第一个比当前元素小的索引
                int left = sta.peek();
                // 右边第一个比当前元素小的索引
                // right = i;
                res = Math.max(res, (i - left - 1) * new_heights[cur]);
           }
           sta.push(i);
       }
       return res;
    }
}

    public ListNode sortList(ListNode head) {
        // 没有节点或只有一个节点
        if (head == null || head.next == null) return head;
        // 找到中间节点
        ListNode mid = findMid(head);
        // 记录下一个节点
        ListNode next = mid.next;
        // 注意要断开链表
        mid.next = null;
        // 排序 [head, mid]
        ListNode left = sortList(head);
        // 排序 [mid.next...]
        ListNode right = sortList(next);
        // 合并链表
        return merge(left, right);
    }

    // 合并两个有序链表
    public ListNode merge(ListNode head1, ListNode head2) {
        ListNode dummy = new ListNode();
        ListNode p = dummy, p1 = head1, p2 = head2;
        while (p1 != null && p2 != null) {
            if (p1.val < p2.val) {
                p.next = p1;
                p1 = p1.next;
            } else {
                p.next = p2;
                p2 = p2.next;
            }
            p = p.next;
        }
        if (p1 != null) {
            p.next = p1;
        }
        if (p2 != null) {
            p.next = p2;
        }
        return dummy.next;
    }

    // 找到链表的中间节点
    public ListNode findMid(ListNode head) {
        // fast = head.next
        // 偶数节点返回中间节点的左边
        ListNode fast = head.next, slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
        }
        return slow;
    }

        boolean isEnd;
        // 字母映射表
        TrieNode[] next;

        root = new TrieNode();

        TrieNode node = root;
        for (int i = 0; i < word.length(); i++) {
            char ch = word.charAt(i);
            if (node.next[ch - 'a'] == null) {
                node.next[ch - 'a'] = new TrieNode();
            } 
            node = node.next[ch - 'a'];
        }
        // 标记结束
        node.isEnd = true;

        TrieNode node = root;
        for (int i = 0; i < word.length(); i++) {
            char ch = word.charAt(i);
            if (node.next[ch - 'a'] == null) {
                return false;
            } 
            node = node.next[ch - 'a'];
        }
        return node.isEnd;

            }

    // 如何由一个小正方形推出大正方形？
    // 当 matrix[i][j] = 1 时，如果它的左、左上、上都存在正方形，那么就可以组成更大的正方形
    public int maximalSquare(char[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        // dp[i][j] 表示以 matirx[i][j] 为右下角的正方形的最大边长
        int[][] dp = new int[m][n];
        int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (i == 0 || j == 0) {
                    dp[i][j] = matrix[i][j] - '0';
                    res = Math.max(dp[i][j], res);
                    continue;
                }
                if (matrix[i][j] == '1') {
                    // 类似【木桶效应】，最大边长取决于最小的边长
                    dp[i][j] = min(dp[i - 1][j], dp[i - 1][j - 1], dp[i][j - 1]) + 1;     
                    res = Math.max(dp[i][j], res);
                } else {
                    // 值为 0 不可能为正方形的右下角
                    dp[i][j] = 0;
                }
            }
        }
        return res * res;
    }
    public int min(int a, int b, int c) {
        return Math.min(a, Math.min(b, c));
    }

                                                                                                      // 时间复杂度 O(nlogn) 遍历 n 个元素，每个元素插入堆的复杂度是 O(logn)
// 空间复杂度 O(n)
class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        int n = nums.length;
        // 元素个数 n - k + 1
        int[] res = new int[n - k + 1];
        int index = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> b[1] - a[1]);
        for (int i = 0; i < n; i++) {
            pq.add(new int[]{i, nums[i]});
            if (i >= k - 1) {
                // 达到首个滑动窗口的右端点
                while (pq.peek()[0] <= i - k) {
                    // 堆顶的元素下标小于滑动窗口的左端点
                    // 删除元素
                    pq.poll();
                }
                res[index++] = pq.peek()[1];
            }
        }
        return res;
    }
}

                                                                                                      解法二：双端队列【单调递减】

                                                                                                      // 时间复杂度 O(n) 遍历 n 个元素，每个元素加入队列的复杂度为 O(1)
// 空间复杂度 O(n)
class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        int n = nums.length;
        // 元素个数 n - k + 1
        int[] res = new int[n - k + 1];
        int index = 0;
        // 双端队列
        Deque<Integer> dq = new ArrayDeque<Integer>();
        for (int i = 0; i < n; i++) {
            while (!dq.isEmpty() && nums[dq.peekLast()] <= nums[i]) {
                // nums[dq.peekLast()] 不是当前窗口的最大值
                // 删除
                dq.pollLast();
            }
            // 加入到队列末尾
            dq.addLast(i);
            // 达到第一个窗口左端点
            if (i >= k - 1) {
                // 此时的队列是【单调递减的】
                // 判断下标是否小于滑动窗口的左端点
                while (!dq.isEmpty() && dq.peekFirst() <= i - k) {
                    dq.pollFirst();
                }
                res[index++] = nums[dq.peekFirst()];
            }
        }
        return res;
    }
}

                right = mid - 1;

class Solution {

    HashSet<String> set = new HashSet<>();
    String s;
    // maxLen：合法子串的最大长度
    // maxScore：最大得分
    int n, maxLen, maxScore;

    public List<String> removeInvalidParentheses(String _s) {
        s = _s;
        n = s.length();
        // 计算左右括号数
        int left = 0, right = 0;
        for (int i = 0; i < n; i++) {
            char ch = s.charAt(i);
            if (ch == '(') left++;
            if (ch == ')') right++;
        }
        // 计算最大得分
        maxScore = Math.min(left, right);
        dfs(0, "", 0);
        return new LinkedList(set);
    }

    /**
        start：开始位置
        track：路径
        score：分数
     */
    public void dfs(int start, String track, int score) {
        // 剪枝逻辑
        if (score < 0 || score > maxScore) return;
        // 遍历到字符串末尾
        if (start == n) {
            // score 为 0，当前路径 >= maxLen 才有效
            if (score == 0 && track.length() >= maxLen) {
                if (track.length() > maxLen) {
                    // 找到更长的合法子串，清空 set
                    set.clear();
                }
                // 记录或更新最大长度
                maxLen = track.length();
                // 加入到 set 中
                set.add(track);
            }
            return;
        }
        // 当前字符
        char ch = s.charAt(start);
        if (ch == '(') {
            // 选择左括号
            dfs(start + 1, track + ch, score + 1);
            // 不选
            dfs(start + 1, track, score);
        } else if (ch == ')') {
            // 选择右括号
            dfs(start + 1, track + ch, score - 1);
            // 不选
            dfs(start + 1, track, score);
        } else {
            // 普通字符
            dfs(start + 1, track + ch, score);
        }
    }
}

        int[][] dp = new int[n + 1][amount + 1];

            dp[0][i] = amount + 1;
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= amount; j++) {
                if (coins[i - 1] > j) {
                    // 不选
                    dp[i][j] = dp[i - 1][j];
                } else {
                    // 完全背包
                    // 不选 || 选
                    dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - coins[i - 1]] + 1);
                }
            }
        }
        return dp[n][amount] == amount + 1? -1 : dp[n][amount];
    }
}

            while (t != 0) {
                // 消除末尾的1
                t = t & (t - 1);
                count++;
            }
            res[i] = count;
        }
        return res;
    }

// 空间复杂度 O(n)
class Solution {
    public int[] topKFrequent(int[] nums, int k) {
        // 使用 map 统计元素出现的个数
        HashMap<Integer, Integer> map = new HashMap<>();
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            map.put(nums[i], map.getOrDefault(nums[i], 0) + 1);
        }
        // 按出现次数构造最小堆
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> map.get(a) - map.get(b));
        // 遍历 map 集合
        for (Integer key : map.keySet()) {
            if (pq.size() < k) {
                // 前 k 个加入堆
                pq.add(key);
            } else {
                if (!pq.isEmpty() && map.get(key) > map.get(pq.peek())) {
                    // 当前元素出现的频率比堆顶的大
                    pq.poll();
                    pq.add(key);
                }
            }
        }
        // 构造返回数据
        // int[] res = new int[k];
        // int idx = 0;
        // while (!pq.isEmpty()) {
        //     res[idx++] = pq.poll();
        // }
        // return res;
        return pq.stream().mapToInt(i -> i).toArray();
    }
}

                                                                                                      class Solution {
    public String decodeString(String s) {
        // 辅助栈，存放倍数 multi 和 结果 res
        Stack<String> sta_res = new Stack<>();
        Stack<Integer> sta_multi = new Stack<>();
        StringBuilder res = new StringBuilder();
        int multi = 0;
        for (char ch : s.toCharArray()) {
            if ('0' <= ch && ch <= '9') {
                // 数字
                multi = multi * 10 + ch - '0';
            } else if (ch == '[') {
                // 数字和res入栈
                sta_multi.push(multi);
                sta_res.push(res.toString());
                // 赋值为初值
                multi = 0;
                res = new StringBuilder();
            } else if (ch == ']') {
                // 出栈
                String last_res = sta_res.pop();
                int cur_multi = sta_multi.pop();
                // 拼接结果
                StringBuilder tmp = new StringBuilder();
                for (int i = 0; i < cur_multi; i++) {
                    tmp.append(res.toString());
                }
                res = new StringBuilder(last_res + tmp.toString());
            } else {
                // 普通字符，直接进栈
                res.append(ch);
            }
        }
        return res.toString();
    }
}

                                                                                                      // 时间复杂度：每个节点只遍历一次,O(N).
// 空间复杂度：开辟了一个hashMap,O(N).
class Solution {
    public int pathSum(TreeNode root, int targetSum) {
        // key 为前缀和，value 是前缀和为 key 的路径数
        HashMap<Long, Integer> preSumMap = new HashMap<>();
        // 前缀和为 0 的路径数为 1
        preSumMap.put(0L, 1);
        return pathSum(root, targetSum, 0, preSumMap);
    }
    public int pathSum(TreeNode root, int target, long curSum, HashMap<Long, Integer> preSumMap) {
        // 结束条件
        if (root == null) return 0;
        // 当前节点需要做的事情
        int res = 0;
        curSum += root.val;
        // curSum - target 就是节点和为 target 的起点
        res += preSumMap.getOrDefault(curSum - target, 0);
        // 前缀和为 curSum 的路径数+1
        preSumMap.put(curSum, preSumMap.getOrDefault(curSum, 0) + 1);
        // 遍历左右节点
        res += pathSum(root.left, target, curSum, preSumMap);
        res += pathSum(root.right, target, curSum, preSumMap);
        // 回到上一层时，撤销当前节点的前缀和路径数
        preSumMap.put(curSum, preSumMap.get(curSum) - 1);
        return res;
    }
}

        int res  = 0;

    public int subarraySum(int[] nums, int k) {

        int n = nums.length;
        int[] preSum = new int[n + 1];
        // 计算前缀和
        for (int i = 1; i <= n; i++) {
            preSum[i] = preSum[i - 1] + nums[i - 1];
        }
        for (int i = 0; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
                // 计算每一个子数组和
                if (preSum[j] - preSum[i] == k) {
                    res++;
                }
            }
        }
        return res;
    }

                                                                                                      class Solution {
    public int findUnsortedSubarray(int[] nums) {
        // 把数组分为三段，左端、中端、右端
        int n = nums.length;
        int max = nums[0];
        int min = nums[n - 1];
        // 中端数组的左右边界
        // 考虑到数组已经升序的情况
        int left = 0, right = -1;
        for (int i = 0; i < n; i++) {
            // 从左往右遍历数组，最后一个小于 max 就是右边界
            if (nums[i] < max) {
                right = i;
            } else {
                max = nums[i];
            }
            // 从右往左遍历数组，最后一个大于 min 就是左边界
            if (nums[n - i - 1] > min) {
                left = n - i - 1;
            } else {
                min = nums[n - i - 1];
            }
        }
        // 返回中端数组长度
        return right - left + 1;
    }
}

    public int leastInterval(char[] tasks, int n) {
        // 用数组统计任务出现的次数
        int[] taskCount = new int[26];
        for (char ch : tasks) {
            taskCount[ch - 'A']++;
        }
        // 计算最大的任务数以及最大的任务数出现的次数
        int max = 0, maxCount = 0;
        for (int count : taskCount) {
            if (max < count) {
                max = count;
                maxCount = 1;
            } else if (max == count) {
                maxCount++;
            }
        }
        // 按 (n + 1, max - 1) 排成一个矩阵，则耗时 (n + 1) * (max - 1) + maxCount
        // 考虑到 n = 0 的情况
        return Math.max((n + 1) * (max - 1) + maxCount, tasks.length);
    }
}
