                                                        # 1.1 动态规划解题套路框架

                                                        1、确定 base case，这个很简单，显然目标金额 amount 为 0 时算法返回 0，因为不需要任何硬币就已经凑出目标金额了。

                                                        2、确定「状态」，也就是原问题和子问题中会变化的变量。由于硬币数量无限，硬币的面额也是题目给定的，只有目标金额会不断地向 base case 靠近，所以唯一的「状态」就是目标金额 amount。

                                                        3、确定「选择」，也就是导致「状态」产生变化的行为。目标金额为什么变化呢，因为你在选择硬币，你每选择一枚硬币，就相当于减少了目标金额。所以说所有硬币的面值，就是你的「选择」。

                                                        4、明确 dp 函数/数组的定义。我们这里讲的是自顶向下的解法，所以会有一个递归的 dp 函数，一般来说函数的参数就是状态转移中会变化的量，也就是上面说到的「状态」；函数的返回值就是题目要求我们计算的量。就本题来说，状态只有一个，即「目标金额」，题目要求我们计算凑出目标金额所需的最少硬币数量。

                                                        所以我们可以这样定义 dp 函数：dp(n) 表示，输入一个目标金额 n，返回凑出目标金额 n 所需的最少硬币数量。

                                                        F(0) = 0，F(1) = 1 F(n) = F(n - 1) + F(n - 2)，其中 n > 1 给定 n ，请计算 F(n) 。

                                                        思路：

                                                        1.base case

                                                        2.f(n) = f(n-1) + f(n-2)

    public int fib(int n) {
        if (n == 0) return 0;
        // 代表 dp[i - 1] 和 dp[i - 2]
        int dp_i_1 = 1, dp_i_2 = 0;
        for (int i = 2; i <= n; i++) {
            int dp_i = dp_i_1 + dp_i_2;
            // 更新
            dp_i_2 = dp_i_1;
            dp_i_1 = dp_i;
        }
        return dp_i_1;
    }
}

                                                        # 322. 零钱兑换open in new window

                                                        给你一个整数数组 coins ，表示不同面额的硬币；以及一个整数 amount ，表示总金额。

                                                        计算并返回可以凑成总金额所需的 最少的硬币个数 。如果没有任何一种硬币组合能组成总金额，返回 -1 。

                                                        你可以认为每种硬币的数量是无限的。

                                                        自底而上思路：

                                                        dp 数组的定义：凑出 i 元至少需要 dp[i] 个硬币

                                                        初始化数组 dp[0...amount] = amount + 1

                                                        1.遍历数组

                                                        2.对数组中的每个元素，求子问题

                                                        3.子问题无解，continue

                                                        4.最少需要的硬币数 = 子问题 + 1，求出子问题中的最小值

                                                        5.返回结果

                                                        class Solution {
    // 定义：凑出 amount 最少需要 coinChange(coins, amount)个硬币
    public int coinChange(int[] coins, int amount) {
        // dp 数组的定义：凑出 i 元至少需要 dp[i] 个硬币
        // 初始化数组 dp[amount + 1] = amount + 1
        int[] dp = new int[amount + 1];
        // amount + 1 就等于初始化为 正无穷
        Arrays.fill(dp, amount + 1);
        // base case
        dp[0] = 0;
        // 1.遍历数组
        for (int i = 1; i < dp.length; i++) {
            // 2.对数组中的每个元素，求子问题
            for (int coin : coins) {
                // 3.子问题无解，continue
                if (i - coin  < 0) {
                    continue;
                }
                // 4.最少需要的硬币数 = 子问题 + 1，求出子问题中的最小值
                dp[i] = Math.min(dp[i], dp[i - coin] + 1);
            }
        }
        // 5.返回结果
        return dp[amount] == amount + 1 ? -1 : dp[amount];
    }
}

                                                        思路：

                                                        1.初始化 res

                                                        2.base case

                                                        3.遍历 conis

                                                        4.求子问题(-1 代表无解)

                                                        5.res = 子问题 + 1，求最小

                                                        class Solution {

    // 备忘录
    int[] memo;

    // 定义：凑出 amount 最少需要 coinChange(coins, amount)个硬币
    public int coinChange(int[] coins, int amount) {
        // 初始化 备忘录
        memo = new int[amount + 1];
        Arrays.fill(memo, -666);
        return dp(coins, amount);
    }

    public int dp(int[] coins, int amount) {    
        // 1.初始化 res
        int res = Integer.MAX_VALUE;
        // 2.base case
        if (amount == 0) return 0;
        if (amount < 0) return -1;
        // 备忘录不为空
        if (memo[amount] != -666) {
            return memo[amount];
        }
        // 3.遍历 conis
        for (int coin : coins) {
            // 4.求子问题
            int sub = dp(coins, amount - coin);
            // 判断是否有解
            if (sub == - 1) {
                continue;
            }
            // 5.子问题中选择最优解，然后加1
            res = Math.min(res, sub + 1);
        } 
        // 加入备忘录
        memo[amount] = (res == Integer.MAX_VALUE) ? -1 : res;
        return memo[amount];
    }
}

                                                        # 300. 最长递增子序列open in new window

                                                        给你一个整数数组 nums ，找到其中最长严格递增子序列的长度。

                                                        子序列 是由数组派生而来的序列，删除（或不删除）数组中的元素而不改变其余元素的顺序。例如，[3,6,2,7] 是数组 [0,3,1,6,2,2,7] 的子序列。

                                                        思路：

                                                        dp数组的定义：以nums[i]结尾的最长子序列的长度

                                                        怎么由 dp[0...4] 推出 dp[5] 呢，首先找到比 nums[5] 小的数，然后他们的子序列长度+1就形成了新的子序列长度，求出最大值

                                                        1.base case 初始化 dp[i] = 1

                                                        2.for 循坏找出比 nums[0...i-1] 中比 nums[i] 小的数

                                                        3.子序列长度+1，求出最大值

                                                        4.求出 dp 数组的最大值，也就是最长子序列长度

                                                        class Solution {
    public int lengthOfLIS(int[] nums) {
        // 1.base case 初始化 dp[i] = 1
        int n = nums.length;
        // dp 数组的定义：以nums[i]结尾的最长子序列的长度
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                // 2.for 循坏找出比 nums[0...i-1] 中比 nums[i] 小的数
                if (nums[i] > nums[j]) {
                    // 3.子序列长度+1，求出最大值
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                }
            }
        }
        // 4.求出 dp 数组的最大值，也就是最长子序列长度
        int res = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            res = Math.max(res, dp[i]);
        }        
        return res;
    }
}

                                                        # 931. 下降路径最小和open in new window

                                                        给你一个 n x n 的 方形 整数数组 matrix ，请你找出并返回通过 matrix 的下降路径 的 最小和

                                                        下降路径 可以从第一行中的任何元素开始，并从每一行中选择一个元素。在下一行选择的元素和当前行所选元素最多相隔一列（即位于正下方或者沿对角线向左或者向右的第一个元素）。具体来说，位置 (row, col) 的下一个元素应当是 (row + 1, col - 1)、(row + 1, col) 或者 (row + 1, col + 1) 。

                                                        自顶而下：

                                                        1.base case: i == 0

                                                        2.状态：下降路径的最小和

                                                        3.选择：matrix[i] [j] 下降路径的最小和可以由 matrix[i - 1][j - 1]、matrix[i - 1][j]、matrix[i - 1][j + 1] 中的最小和加上当前元素确定

                                                        4.dp函数：dp(int[][] matrix, int i, int j)：从第一行到 matrix[i] [j] 的下降路径最小和

                                                        思路：

                                                        1.索引下标越界检查，返回 99999，由 1 <= n <= 100 和 -100 <= matrix[i] [j] <= 100 决定

                                                        2.base case：i == 0，返回 matrix[0] [j]

                                                        3.确定 matrix[i] [j] 下降路径的最小和

                                                        4.备忘录

                                                        class Solution {

    // 备忘录
    int[][] memo;

    public int minFallingPathSum(int[][] matrix) {    
        int n = matrix.length;
        // 初始化备忘录
        memo = new int[n][n];
        for (int i = 0; i < n; i++) {
            Arrays.fill(memo[i],66666);
        }
        int res = Integer.MAX_VALUE;
        // 最小和可能在 最后一行的任意一列
        for (int i = 0; i < n; i++) {
            res = Math.min(res, dp(matrix, n - 1, i));
        }
        return res;
    }
    
    // 定义：从第一行到 matrix[i] [j] 的下降路径最小和
    public int dp(int[][] matrix, int i, int j) {
        // 1.索引下标越界检查，返回 99999，由 1 <= n <= 100  和  -100 <= matrix[i] [j] <= 100 决定
        if (i < 0 || j < 0 || i >= matrix.length || j >= matrix.length) return 99999;
        // 2.base case：i == 0，返回 matrix[0] [j]
        if (i == 0) return matrix[0][j];
        // 查备忘录
        if (memo[i][j] != 66666) return memo[i][j];
        // 3.状态：确定 matrix[i] [j] 下降路径的最小和
        memo[i][j] = matrix[i][j] + min(
                                    dp(matrix, i - 1, j - 1),
                                    dp(matrix, i - 1, j),
                                    dp(matrix, i - 1, j + 1)
                                    );
        // 4.备忘录
        return memo[i][j];
    }
    public int min(int a, int b, int c) {
        return Math.min(a, Math.min(b, c));
    }
}

                                                        状态：下降路径最小和

                                                        选择：左上、上、右上

                                                        dp数组的定义：dp[i] [j] 表示从第一行到 matrix[i] [j] 的下降路径最小和

                                                        base case: dp[0] [...] = matrix[0] [...]

                                                        思路：

                                                        1.定义dp数组

                                                        2.base case：dp[0] [...] = matrix[0] [...]

                                                        class Solution {
    public int minFallingPathSum(int[][] matrix) {
        // 1.定义dp数组
        int n = matrix.length;
        int[][] dp = new int[n][n];
        // 2.base case：dp[0] [...] = matrix[0] [...]
        for (int i = 0; i < n; i++) {
            dp[0][i] = matrix[0][i];
        }
        // 3.遍历数组
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < n; j++) {
                // 4.处理越界问题
                if (j == 0) {
                    dp[i][j] = matrix[i][j] + Math.min(dp[i - 1][j], dp[i - 1][j + 1]);
                } else if (j == n - 1) {
                    dp[i][j] = matrix[i][j] + Math.min(dp[i - 1][j - 1], dp[i - 1][j]);
                } else {
                    // 5.做选择
                    dp[i][j] = matrix[i][j] + min(dp[i - 1][j - 1], dp[i - 1][j], dp[i - 1][j + 1]);
                }
            }
        }
        // 6.返回结果
        int res = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            res = Math.min(res, dp[n - 1][i]);
        }
        return res;
    }
   public int min(int a, int b, int c) {
       return Math.min(a, Math.min(b, c));
   }
}

                                                        # 72. 编辑距离open in new window

                                                        给你两个单词 word1 和 word2， 请返回将 word1 转换成 word2 所使用的最少操作数 。

                                                        你可以对一个单词进行如下三种操作：

                                                        插入一个字符 删除一个字符 替换一个字符

                                                        思路：

                                                        dp数组的定义：dp[i] [j] 存储 s1[0...i - 1] 和 s2[0...j-1] 的编辑距离

                                                        1.base case

                                                        2.s1[i - 1] 和 s2[j - 1] 相等,skip

                                                        3.否则 增删改 + 1，求最小值

                                                        class Solution {
    public int minDistance(String word1, String word2) {
        int m = word1.length(), n = word2.length();
        // dp[i][j] 存储 s1[0...i - 1] 和 s2[0...j-1] 的编辑距离
        int[][] dp = new int[m + 1][n + 1];
        // 1.base case，s1 或 s2 为空串，全增加或全删除
        for (int i = 0; i <= m; i++) {
            dp[i][0] = i;
        }
        for (int i = 0; i <= n; i++) {
            dp[0][i] = i;
        }
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                // 2.s1[i - 1] 和 s2[j - 1] 相等,skip
                if (word1.charAt(i - 1) == word2.charAt(j - 1)) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    // 3.否则 增删替 + 1，求最小值
                    dp[i][j] = 1 + min(
                       	// 插入	
                        dp[i][j - 1],
                        // 删除
                        dp[i - 1][j],
                        // 替换
                        dp[i - 1][j - 1]
                    );
                }
               
            }
        }
        return dp[m][n];
    }
    public int min(int a, int b, int c) {
        return Math.min(a, Math.min(b, c));
    }
}

                                                        # 53. 最大子数组和open in new window

                                                        给你一个整数数组 nums ，请你找出一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和。

                                                        子数组 是数组中的一个连续部分。

                                                        状态：连续子数组的最大和

                                                        选择：nums[i] 的选择是 nums[i] 和 dp[i - 1] + nums[i]

                                                        base case: dp[0] = nums[0]

                                                        dp数组的定义：以 nums[i] 结尾的连续子数组的最大和

                                                        思路：

                                                        1.base case：dp[0] = nums[0]

                                                        2.遍历数组，从 dp[i - 1] + nums[i] 和 nums[i] 中选择最大值

                                                        3.返回 dp 数组中的最大值

                                                        class Solution {
    public int maxSubArray(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        // 1.base case：dp[0] = nums[0]
        dp[0] = nums[0];
        // 2.遍历数组，从 dp[i - 1] + nums[i] 和 nums[i] 中选择最大值
        int res = dp[0];
        for (int i = 1; i < n; i++) {
            dp[i] = Math.max(dp[i - 1] + nums[i], nums[i]);
            // 3.返回 dp 数组中的最大值
            res = Math.max(res, dp[i]);
        }
        return res;
    }
}

    public int maxSubArray(int[] nums) {

        int res = dp_0;

            dp_1 = Math.max(dp_0 + nums[i], nums[i]);
            dp_0 = dp_1;
            // 3.返回 dp 数组中的最大值
            res = Math.max(res, dp_1);
        }
        return res;
    }
}

                                                        1.构造前缀和数组

                                                        2.遍历数组

                                                        3.找出 min(preSum[0...i])

                                                        4.求 preSum[i + 1] - min(preSum[0...i]) 的最大值

                                                        class Solution {
    public int maxSubArray(int[] nums) {
        // 1.构造前缀和数组
        int n = nums.length;
        int[] preSum = new int[n + 1];
        preSum[0] = 0;
        for (int i = 1; i <= n; i++) {
            preSum[i] = preSum[i - 1] + nums[i - 1];
        }
        // 2.遍历数组
        int minVal = Integer.MAX_VALUE;
        int res = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            // 3.找出 min(preSum[0...i])
            minVal = Math.min(minVal, preSum[i]);
            // 4.求 preSum[i + 1] - min(preSum[0...i]) 的最大值
            res = Math.max(res, preSum[i + 1] - minVal);
        }
        return res;
    }
}

        // 2.base case：dp[0] [...] = dp[...] [0] = 0
        // 3.遍历数组
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                // 4.选择   
                // 4.1.s1[i - 1] == s2[j - 1]
                if (text1.charAt(i - 1) == text2.charAt(j - 1)) {

    // 1.定义二维数组 dp[W + 1] [N + 1]
    int[][] dp = new int[W + 1][N + 1];
    // 2.base case 
    // 3.遍历物品，状态，做选择
    for (int i = 1; i <= N; i++) {
        for (int w = 1; w <= W; w++) {
            // 3.1 w < wt[i - 1],只能选择不装 dp[i] [w] = dp[i - 1][w]
            if (w < wt[i - 1]) {
                dp[i][w] = dp[i - 1][w];
            } else {
                // 3.2 w >= wt[i - 1]，在不装 dp[i - 1] [w] 和 装 val[i - 1] + dp[i - 1][w - wt[i - 1]] 之间选择最大值
                dp[i][w] = Math.max(dp[i - 1][w], val[i - 1] + dp[i - 1][w - wt[i - 1]]);	
            } 
        }
    }
    // 4.返回结果 dp[N][W]
    return dp[N][W];
}

                                                        # 416. 分割等和子集open in new window

                                                        给你一个 只包含正整数 的 非空 数组 nums 。请你判断是否可以将这个数组分割成两个子集，使得两个子集的元素和相等。

                                                        示例 1：

                                                        输入：nums = [1,5,11,5]
输出：true
解释：数组可以分割成 [1, 5, 5] 和 [11] 。

                                                        转化为背包问题：

                                                        计算数组的和 sum，给你一个容量为 sum/2 的背包，有 n 个物品，每个物品的重量为 nums[i]，是否存在一种装法，使得背包恰好被装满?

                                                        状态：容量 和 可选择的物品

                                                        选择：选 或者 不选

                                                        dp数组的定义：dp[i] [j] = x,在容量为 j 的情况下，前 i 个物品是否恰好可以装满背包

                                                        base case：没有物品 dp[0] [...] = false, 背包没有空间，等于已经装满 dp[...] [0] = true

                                                        思路：

                                                        1.计算数组的和 sum

                                                        sum 为奇数，不可能划分成两个和相等的集合

                                                        sum = sum / 2

                                                        2.dp[n + 1] [sum + 1]

                                                        base case：dp[...] [0] = true

                                                        3.遍历

                                                        4.选择

                                                        4.1 背包容量小于 nums[i - 1]，只能选择不装 dp[i] [sum] = dp[i - 1] [sum]

                                                        4.2 装 dp[i] [sum] = dp[i - 1] [sum - nums[i - 1]] 或者 不装

                                                        5.返回结果 dp[n] [sum]

                                                        class Solution {
    public boolean canPartition(int[] nums) {
        // 1.计算数组的和 sum
        int sum = 0;
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
        }
        // sum 为奇数，不可能划分成两个和相等的集合
        if (sum % 2 == 1) {
            return false;
        }
        sum = sum / 2;
        // 2.dp[n + 1] [sum + 1]
        boolean[][] dp = new boolean[n + 1][sum + 1];
        // base case：dp[...] [0] = true，背包没有空间，等于已经装满
        for (int i = 0; i <= n; i++) {
            dp[i][0] = true;
        }
        // 3.遍历
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= sum; j++) {
                // 4.选择
                // 4.1 背包容量小于 nums[i - 1]，只能选择不装 dp[i] [sum] = dp[i - 1] [sum]
                if (j < nums[i - 1]) {
                    dp[i][j] = dp[i - 1][j];
                } else {
                    // 4.2 装 dp[i] [sum] = dp[i - 1] [sum - nums[i - 1]] 或者 不装 
                    dp[i][j] = dp[i - 1][j - nums[i - 1]] || dp[i - 1][j];
                }
            }
        }
        // 5.返回结果 dp[n] [sum]
        return dp[n][sum];
    }
}

    public int minPathSum(int[][] grid) {
    // 1.定义 dp 数组
    int m = grid.length;
    int n = grid[0].length;
    int[][] dp = new int[m][n];
    // 2.base case：dp[0] [0] = grid[0] [0]
    dp[0][0] = grid[0][0];
    // i == 0, dp[0] [j] = dp[0] [j - 1] + grid[0] [j]
    for (int i = 1; i < n; i++) {

    }
    // j == 0, dp[i] [0] = dp[i - 1] [0] + grid[i] [0]

        dp[j][0] = dp[j - 1][0] + grid[j][0];
    }
    // 3.遍历数组
    for (int i = 1; i < m; i++) {
        for (int j = 1; j < n; j++) {
            // 4.做选择 dp[i] [j] = min(dp[i] [j - 1], dp[i - 1] [j]) + grid[i] [j]
            dp[i][j] = Math.min(dp[i][j - 1], dp[i - 1][j]) + grid[i][j];
        }
    }
    // 5.返回结果 dp[m - 1] [n - 1]

    // 值到索引的映射
    Map<Character, List<Integer>> map = new HashMap<>();
    // 备忘录
    int[][] memo;

    public int findRotateSteps(String ring, String key) {
        // 初始化memo为0
        int m = ring.length();
        int n = key.length();
        memo = new int[m][n];
        // 1.定义一个映射 map
        for (int i = 0; i < m; i++) {
            char c = ring.charAt(i);
            // 字符可能有多个
            map.putIfAbsent(c, new ArrayList<Integer>());
            map.get(c).add(i);
        }
        // 2.调用函数 dp(ring, 0, key, 0)
        return dp(ring, 0, key, 0);

    }
    // 表示当前指针为 ring[i]，输入 key[j...] 的最小步数
    public int dp(String ring, int i, String key, int j) {
        // 1.base case：结束
        if (j == key.length()) return 0;
        // 查备忘录
        if (memo[i][j] != 0) return memo[i][j];
        int n = ring.length();
        int res = Integer.MAX_VALUE;
        // 2.遍历所有的索引
        for (int k : map.get(key.charAt(j))) {
            // 3.得到转动的次数
            int count = Math.abs(k - i); // 记得加绝对值
            // 4.选择顺时针还是逆时针
            count = Math.min(count, n - count);
            // 5.继续调用 dp，指针转到 ring[k]，输入下一个值 key[j + 1...]
            int subProblem = dp(ring, k, key, j + 1); // 注意当前的位置是 k
            // 6.选择整体上最小的步数，+ 1 是因为还要按下中心按钮
            res = Math.min(res, count + 1 + subProblem);
        }
        memo[i][j] = res;
        return res;
    }
}

    // 记录所有的入度
    Map<Integer, List<int[]>> map = new HashMap<>();
    // 出发点
    int src;
    // 备忘录
    int[][] memo;

    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        this.src = src;
        // 将中转站转化为中转边数
        k++;
        // 初始化备忘录
        memo = new int[n][k + 1];
        // 1.map 记录一个节点所有的入度以及权重
        for (int[] flight: flights) {
            int from = flight[0];
            int to = flight[1];
            int price = flight[2];
            map.putIfAbsent(to, new ArrayList<>());
            map.get(to).add(new int[]{from, price});
        }
        // 2.调用dp(dst, k)
        return dp(dst, k);
    }
    // 从 src 出发，在k步之内（一步就是一边）到达节点s的最少成本
    public int dp(int s, int k) {
        // 1.base case
        if (s == this.src) {
            // 出发点已经在目的地
            return 0;
        }
        if (k == 0) {
            // 没有中转机会了
            return -1;
        }
        if (memo[s][k] != 0) return memo[s][k];
        int res = Integer.MAX_VALUE;
        // 2.遍历所有的入度
        if (map.containsKey(s)) {
            List<int[]> in = map.get(s);
            for (int[] arr : in) {
                int from = arr[0];
                int price = arr[1];

                if (subProblem != -1) { // 注意
                    // 5.最少成本 = subProblem + price，选最小值
                    res = Math.min(res, subProblem + price);
                }
            }
        }
        // 6.判断 res 的合法性
        memo[s][k] = res == Integer.MAX_VALUE ? -1 : res; /
        return memo[s][k];
    }

                                                        自顶而下思路：

                                                        1.s 和 p 匹配

                                                        1.1 p 的下一个是 *，匹配0个或多个

                                                        1.2 只能匹配一次

                                                        2.s 和 p 不匹配

                                                        2.1 p 的下一个是 *，匹配 0 个

                                                        2.2 不匹配

                                                        状态：i 和 j

                                                        选择：匹配多少个字符

                                                        dp函数的定义：dp(s,i,p,j) 表示 s[i...] 是否可以匹配 p[j...]

                                                        base case:

                                                        1.j == n,判断 i 是否到末尾

                                                        2.i == m

                                                        2.1 n - j 是否是偶数

                                                        2.2 判断字符串p 是否是 x*y*z* 的形式

                                                        备忘录

                                                        dp(s, i, p, j + 2) // 1

                                                        dp(s, i + 1, p, j) // 2

                                                        dp(s, i + 1, p, j + 1) // 3

                                                        变成 dp(s, i + 2, p, j + 2) 1->2->2 || 3->3

                                                        class Solution {

    // 备忘录
    Map<String, Boolean> memo = new HashMap<>();

    public boolean isMatch(String s, String p) {
        return dp(s, 0, p, 0);
    }
    // dp函数的定义：dp(s,i,p,j) 表示 s[i...] 是否可以匹配 p[j...]
    public boolean dp(String s, int i, String p, int j) {
        int m = s.length(), n = p.length();
        // base case: 
        // 1.j == n,判断 i 是否到末尾
        if (j == n) {
            return i == m;
        }
        // 2.i == m
        if (i == m) {
            // 2.1 n - j 是否是偶数
            if ((n - j) % 2 == 1) {
                return false;
            }
            // 2.2 判断字符串p 是否是 `x*y*z*` 的形式
            for(; j + 1 < n; j += 2) {
                if (p.charAt(j + 1) != '*') {
                    return false;
                }
            }
            return true;
        }
        // 查备忘录
        String key = i + "," + j;
        if (memo.containsKey(key)) {
            return memo.get(key);
        }
        boolean res = false;

        // 3.s 和 p 匹配
        if (s.charAt(i) == p.charAt(j) || p.charAt(j) == '.') {
            // 3.1 p 的下一个是 *，匹配0个或多个
            if (j + 1 < n && p.charAt(j + 1) == '*') {
                res =  dp(s, i, p, j + 2) // 匹配0个
                                        || dp(s, i + 1, p, j); // 匹配多个
            } else {
                // 3.2 只能匹配一个
                res =  dp(s, i + 1, p, j + 1);
            }
        } else {
            // 4.s 和 p 不匹配
            // 4.1 p 的下一个是 *，匹配 0 个
            if (j + 1 < n && p.charAt(j + 1) == '*') {
                res =  dp(s, i, p, j + 2);
            } else {
                // 4.2 不匹配
                res =  false;
            }
        }
        memo.put(key, res);
        return res;
    }
}

    // 备忘录
    Map<TreeNode, Integer> memo = new HashMap<>();

    public int rob(TreeNode root) {
        if (root == null) return 0;
        // 查备忘录
        if (memo.containsKey(root)) return memo.get(root);
        // 抢：去下下家
        int rob = root.val + (root.left == null ? 0 : rob(root.left.left) + rob(root.left.right)) 
                            + (root.right == null ? 0 : rob(root.right.left) + rob(root.right.right));
        // 不抢：去下家
        int no_rob = rob(root.left) + rob(root.right);
        int res =  Math.max(rob, no_rob);
        memo.put(root, res);
        return res;
    }
}

                                                        状态：天数、交易次数、当前是否持有股票

                                                        选择：sell、buy、rest（无操作）

                                                        dp[i] [k] [0/1]：在第 i 天，交易次数为 k，当前是否持有股票（0/1）的最大利润

                                                        2.base case：i == -1

                                                        3.做选择

                                                        3.1 dp[i] [0]

                                                        3.2 dp[i] [1]

                                                        4.返回结果 dp[n - 1] [0]

                                                        class Solution {
    public int maxProfit(int[] prices) {
        // 1.新建dp数组,dp[i] [k] [0/1]表示在第 i 天，交易次数为 k，当前是否持有股票（0/1）的最大利润
        int n = prices.length;
        int[][] dp = new int[n][2];
        for (int i = 0; i < n; i++) {
            // 2.base case：i == -1
            if (i - 1 == -1) {
                // 还没开始而且没有股票
                dp[i][0] = 0;
                // 还没开始但是已经有了股票
                dp[i][1] = -prices[0];
                continue;

            // 3.做选择，这道题的 k == 1
            // 3.1 dp[i][k][0] = max(dp[i - 1][k][0], dp[i - 1][k][1] + prices[i])
            // (rest,sell)

            // (rest,buy)
            dp[i][1] = Math.max(dp[i - 1][1], -prices[i]);

    }

        if (n == 0) return 0;
        int[][][] dp = new int[n][k + 1][2];
        for (int i = 0; i < n; i++) {
            for (int j = k; j >= 1; j--) {
                // base case
                if (i - 1 == -1) {
                    dp[i][j][0] = 0;
                    dp[i][j][1] = -prices[i];
                    continue;
                }
                // 选择
                dp[i][j][0] = Math.max(dp[i - 1][j][0], dp[i - 1][j][1] + prices[i]);
                dp[i][j][1] = Math.max(dp[i - 1][j][1], dp[i - 1][j - 1][0] - prices[i]);
            }
        }
        return dp[n - 1][k][0];
    }

                                                        选择：选择最左边的石子堆，选择最右边的石子堆

                                                        i == j 只有一堆石子

                                                        2.定义二维dp数组：dp[i] [j].fir 表示面对石子堆 piles[i...j] 先手最多可以得到的石子数

                                                        3.初始化dp数组

                                                        4.base case: i == j，只有一堆石子

                                                        5.倒着遍历数组（因为知道 dp[i + 1] [j] 和 dp[i] [j - 1]）

                                                        6.做选择

                                                        6.1 先手选择最左边

                                                        dp[i] [j].fir = piles[i] + dp[i + 1] [j].sec

                                                        后手下一轮变成先手

                                                        dp[i] [j].sec = dp[i + 1] [j].fir

                                                        6.2 先手选择最右边

                                                        dp[i] [j].sec = piles[j] + dp[i] [j - 1].sec,

                                                        后手下一轮变成先手

                                                        dp[i] [j].sec = dp[i] [j - 1].fir

                                                        7.返回结果 dp[0] [n - 1].fir - dp[0] [n - 1].sec

    // 1.定义 Pair类 表示先手和后手
    class Pair {
        int fir;
        int sec;
        public Pair(int fir, int sec) {
            this.fir = fir;
            this.sec = sec;
        }
    }

    public boolean stoneGame(int[] piles) {
        // 2.定义二维dp数组：dp[i] [j].fir 表示面对石子堆 piles[i...j] 先手最多可以得到的石子数  
        int n = piles.length;
        Pair[][] dp = new Pair[n][n];
        // 3.初始化dp数组
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                dp[i][j] = new Pair(0, 0);
            }
        }
        // 4.base case: i == j，只有一堆石子
        for (int i = 0; i < n; i++) {
            dp[i][i].fir = piles[i];
            dp[i][i].sec = 0;
        }
        // 5.倒着遍历数组
        for (int i = n - 2; i >= 0; i--) {
            for (int j = i + 1; j < n; j++) {
                // 6.做选择
                int left = piles[i] + dp[i + 1][j].sec;
                int right = piles[j] + dp[i][j - 1].sec;
                // 6.1 先手选择最左边 dp[i] [j].fir = dp[i + 1] [j].sec, dp[i] [j].sec = dp[i + 1] [j].fir
                if (left > right) {
                    // 先手选择数量多的，后手的选择随之改变
                    dp[i][j].fir = left;
                    // 后手之后变成了先手
                    dp[i][j].sec = dp[i + 1][j].fir;
                } else {
                    // 6.2 先手选择最右边 dp[i] [j].sec = dp[i] [j - 1].sec, dp[i] [j].sec = dp[i] [j - 1].fir
                    dp[i][j].fir = right;
                    dp[i][j].sec = dp[i][j - 1].fir;
                }
            }
        }
        // 7.返回结果 dp[0] [n - 1].fir - dp[0] [n - 1].sec
        return dp[0][n - 1].fir - dp[0][n - 1].sec >= 0;
    }
}

                f_end = inter[1];
            }
        }
        return count;
    }

        while (i < n && clips[i][0] <= curEnd) {

            curEnd = nextEnd;

                                                        1.记录每一次跳跃的终点(end)

                                                        2.对于这个闭区间，更新可以跳到的最远距离

                                                        3.i 走到区间末尾并且还没有到数组末尾时，次数+1，更新区间

                                                        4.返回结果，判断 end 和 n - 1

                                                        class Solution {
    public int jump(int[] nums) {
        // 1.记录每一次终点(end)
        int end = 0;
        // 最远距离
        int farthest = 0;
        int n = nums.length;
        int res = 0;
        // 注意：i < n - 1
        // 若 i == n - 1 就等于已经走到了最后一个位置
        for (int i = 0; i < n - 1; i++) {
            // 2.对于这个闭区间[i, end]，更新可以跳到的最远距离
            farthest = Math.max(farthest, i + nums[i]);
            // 3.i 走到区间末尾并且还没有到数组末尾时，次数+1，更新区间
            if (i == end) {
                res++;
                // 相当于跳跃一次
                end = farthest;
            }
        }
        return res;
    }
}

        // 1.判断是否有解（总油量减去耗油量）
        int sum = 0;
        int n = gas.length;
        for (int i = 0; i < n; i++) {
            sum += gas[i] - cost[i];
        }
        if (sum < 0) {
            // 无解
            return -1;
        }
        // 油箱容量
        int tank = 0;
        // 起点
        int start = 0;
        // 2.遍历加油站
        for (int i = 0; i < n; i++) {
            // 3.计算油箱容量 tank
            tank += gas[i] - cost[i];
            // 4.判断 tank 
            if (tank < 0) {
                // 5.tank < 0，则说明 i + 1 不可达，起点从 i + 1 开始
                start = (i + 1) % n;
                // 油箱容量置为0
                tank = 0;
            }
        }
        // 6.返回结果 start       
        return start;
    }
